DECLARE myStr : STRING
DECLARE myChar : CHAR
DECLARE finalStr : STRING
DECLARE nextChar : CHAR
DECLARE Index : INTEGER

myStr = ""
myChar = ''
finalStr = ""
nextChar = ''
Index = 0

OUTPUT "Enter string: "
INPUT myStr

OUTPUT "Enter character to remove: "
INPUT myChar

FOR Index = 1 to LEN(myStr)
    nextChar = MID(myStr, Index, 1)
    IF nextChar <> myChar
      THEN
        finalStr = finalStr & nextChar
    ENDIF
ENDFOR

OUTPUT "Final string is = ", finalStr
