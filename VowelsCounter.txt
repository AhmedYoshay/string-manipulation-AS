DECLARE myStr : STRING
DECLARE myChar : CHAR
DECLARE Counter : INTEGER
DECLARE vCount : INTEGER

CONSTANT Vowels = "aeiou"

myStr = ""
myChar = ''
Counter = 0
vCount = 0

OUTPUT "Enter string to look for vowels: "
INPUT myStr

FOR Counter = 1 TO LEN(myStr)
    myChar = MID(myStr, Counter, 1)
    IF LOCATE(Vowels, myChar) > 0
      THEN
	vCount = vCount + 1
    ENDIF
ENDFOR

OUTPUT "Number of vowels in given string = ", Vcount
